void repairTheStreet() {
    repeat(10) {
        if(rightIsClear()) {
            fillHole();
        }
        if(frontIsClear()) {
            moveForward();
        }
    }
}

void fillHole() {
    turnRight();
    moveForward();
    dropBeeper();
    turnAround();
    moveForward();
    turnRight();
}
